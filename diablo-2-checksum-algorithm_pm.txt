#!/usr/bin/perl
               
use v5.16; 

package a1;
use warnings;
use strict;

use parent qw(Exporter); our @EXPORT = qw//;
our @EXPORT_OK = qw/a_ b_ c_/; 
our $all = undef;


sub set_all(){
  my $D2sDataFile = $_;                                                                                    
  my $cached = $_;
  if(! -f $D2sDataFile){
    die "$D2sDataFile does not exist! What do you want to change?";
  }
  {
    local $/;
    open(my ${f}, '<', $D2sDataFile) or die $!;
    $all = <$f>;
    close(${f});
  }
  $_ = $cached;
}

sub a_(){       
  my $cached = $_;               
  if(!defined($all)){
    set_all();
  }
  if(!defined($all)){die '';} 
  say uc(unpack('H*', substr($all, 12, 4)));
  $_ = $cached;
}

sub b_(){
  my $cached = $_;
  if( !defined($all)){
    set_all();
  }
  if(!defined($all)){die '';} 
  my $c = 0;
  my $addr = 0;
  my @cs = unpack('C*', $all);
  for my $ct(@cs){
    $ct += $c >> 31 & 1;
    $c = (($addr >> 2 == 3) ? 0 : $ct) + ($c << 1);
    $addr++;
  }
  my $ret = ($c & 0xff000000) >> 24 | ($c & 0xff0000) >> 8 | ($c & 0xff00) << 8 | ($c & 0xff) << 24;
  printf('%08X' . "\n", $ret);
  undef $ret;
  $_ = $cached;
}

sub c_(){
  my $cached = $_;
  my $D2sDataFile = $_;
  say "xxd '$D2sDataFile'";           
  system("xxd '$D2sDataFile' | head -8");
  $_ = $cached;
}
1;
